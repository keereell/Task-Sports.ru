def ByType(type_):
    modes = df[df['Type']==type_].mode().drop('Type', axis=1).iloc[0]
    return df[df['Type']==type_].describe(percentiles=[]).drop(['count', 'std'], axis=0).append(modes).rename(index={"50%": "median", 0: "mode"})
